
expr : ACONST
     | LCONST
     | ID
     | INCDEC ID
     | NOT expr
     | expr GTLT expr
     | expr OR expr
     | ID OPEN expr COMA expr CLOSE
     | ID OPEN expr CLOSE
     | GET
     | SIZE1 ID
     | SIZE2 ID expr
     | MOVE
     | PUSH
     | OPEN expr CLOSE


lexpr : LCONST
      | ACONST GTLT ACONST
      | NOT lexpr
      | lexpr OR lexpr
      | OPEN lexpr CLOSE

expr : ID
     | INCDEC ID
     | expr GTLT ACONST
     | ACONST GTLT expr
     | expr GTLT expr
     | NOT expr
     | expr OR lexpr
     | lexpr OR expr
     | expr OR expr
     | ID OPEN expr COMA expr CLOSE
     | ID OPEN ACONST COMA expr CLOSE
     | ID OPEN expr COMA ACONST CLOSE
     | ID OPEN ACONST COMA ACONST CLOSE
     | ID OPEN expr CLOSE
     | ID OPEN ACONST CLOSE
     | SIZE1 ID
     | SIZE2 ID expr
     | SIZE2 ID ACONST
     | GET
     | MOVE
     | PUSH
     | OPEN expr CLOSE
