prog : prog line ES
     | prog fdecl ES
     | prog ES
     | 

line : sent
     | decl
     | cdecl
     | logic
     | fcall
     | group
     | 

sent : ID EQ expr
     | expr

logic : cond
      | loop

cond : IF expr sent else
     | IF expr group else

else : ELSE sent
     | ELSE group
     | 

loop : WHILE expr

group : FOPEN lines FCLOSE

lines : lines line ES
      |
